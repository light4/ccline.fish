```bash
# 搜索有 discussions 的高星 shell/terminal/AI 相关 repo
gh api graphql -f query='{search(query:"topic:zsh stars:>5000",type:REPOSITORY,first:20){nodes{...on Repository{nameWithOwner stargazerCount hasDiscussionsEnabled}}}}'

# 搜索某个 repo 的 discussion 分类 ID
gh api graphql -f query='{repository(owner:"OWNER",name:"REPO"){id discussionCategories(first:10){nodes{id name}}}}'

# 创建 discussion（记得换 repositoryId 和 categoryId）
gh api graphql -f query='mutation{createDiscussion(input:{repositoryId:"...",categoryId:"...",title:"...",body:"..."}){discussion{url number}}}'
```
